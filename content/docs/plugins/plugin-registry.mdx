## Installation & Management

### CLI Commands

```bash
# Browse available plugins
elizaos plugins list

# Search for specific plugins
elizaos plugins search weather

# Install a plugin
elizaos plugins add weather

# Install specific version
elizaos plugins add weather@1.2.0

# Install from GitHub
elizaos plugins add https://github.com/user/my-plugin

# Remove a plugin
elizaos plugins remove weather

# Update all plugins
elizaos plugins update

# Show installed plugins
elizaos plugins installed
```

### Configuration Management

Create a `.env` file for plugin configuration:

```env
# Core Configuration
DATABASE_URL=postgresql://user:pass@localhost/db

# AI Models
OPENAI_API_KEY=sk-your-openai-key
ANTHROPIC_API_KEY=sk-ant-your-key
GOOGLE_API_KEY=your-google-key

# Communication
DISCORD_BOT_TOKEN=your-discord-token
TELEGRAM_BOT_TOKEN=your-telegram-token
SLACK_BOT_TOKEN=xoxb-your-slack-token

# Blockchain
ETHEREUM_RPC_URL=https://mainnet.infura.io/v3/your-key
SOLANA_RPC_URL=https://api.mainnet-beta.solana.com

# Services
WEATHER_API_KEY=your-weather-key
NEWS_API_KEY=your-news-key
GITHUB_TOKEN=ghp_your-github-token
```

### Character Configuration

Add plugins to your agent character:

```typescript
// character.ts
import { Character } from '@elizaos/core';

export const character: Character = {
  name: 'MyAgent',
  description: 'An AI agent with comprehensive capabilities',
  plugins: [
    // Core plugins (required)
    '@elizaos/plugin-bootstrap',
    '@elizaos/plugin-sql',
    
    // AI model plugins
    '@elizaos/plugin-openai',
    '@elizaos/plugin-anthropic',
    
    // Communication plugins
    '@elizaos/plugin-discord',
    '@elizaos/plugin-telegram',
    '@elizaos/plugin-twitter',
    
    // Blockchain plugins
    '@elizaos/plugin-solana',
    '@elizaos/plugin-ethereum',
    
    // Utility plugins
    '@elizaos/plugin-browser',
    '@elizaos/plugin-github',
    '@elizaos/plugin-giphy'
  ],
  
  // Plugin-specific settings
  settings: {
    // OpenAI configuration
    openai: {
      model: 'gpt-4',
      temperature: 0.7,
      max_tokens: 2000
    },
    
    // Discord configuration
    discord: {
      enableSlashCommands: true,
      moderationEnabled: true
    },
    
    // Solana configuration
    solana: {
      network: 'mainnet',
      commitment: 'confirmed'
    }
  }
};
```

## Plugin Development

### Creating Custom Plugins

Use the CLI to create a new plugin:

```bash
# Create a new plugin
elizaos create my-plugin -t plugin

# Navigate to plugin directory
cd my-plugin

# Start development server
elizaos dev

# Run tests
elizaos test

# Build for production
elizaos build
```

### Plugin Structure

```
my-plugin/
├── src/
│   ├── actions/          # Agent actions
│   ├── providers/        # Context providers
│   ├── services/         # Background services
│   ├── evaluators/       # Message evaluators
│   ├── index.ts          # Main export
│   └── plugin.ts         # Plugin definition
├── tests/
│   ├── actions.test.ts
│   ├── providers.test.ts
│   └── integration.test.ts
├── images/
│   └── plugin-icon.png
├── package.json
├── tsconfig.json
└── README.md
```

### Plugin Interface

```typescript
// src/plugin.ts
import { Plugin } from '@elizaos/core';

export const myPlugin: Plugin = {
  name: 'my-plugin',
  description: 'Custom plugin functionality',
  
  // Configuration
  config: {
    API_KEY: process.env.MY_API_KEY,
    BASE_URL: process.env.MY_BASE_URL || 'https://api.example.com'
  },
  
  // Initialization
  init: async (config, runtime) => {
    // Initialize plugin
    if (!config.API_KEY) {
      throw new Error('API_KEY is required');
    }
  },
  
  // Components
  actions: [
    // Define actions
  ],
  
  providers: [
    // Define providers
  ],
  
  services: [
    // Define services
  ],
  
  evaluators: [
    // Define evaluators
  ]
};
```

## Publishing Plugins

### Preparation

Before publishing:

1. **Test Thoroughly**: Run all tests and ensure functionality
2. **Documentation**: Write comprehensive README and examples
3. **Assets**: Include plugin icon and screenshots
4. **Validation**: Use `elizaos plugins validate` to check structure

### Publishing Process

```bash
# Validate plugin structure
elizaos plugins validate

# Test publishing (dry run)
elizaos plugins publish --dry-run

# Publish to registry
elizaos plugins publish

# Publish to npm only
elizaos plugins publish --npm-only
```
